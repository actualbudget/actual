> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 52-second Remotion video showcasing 7 key features of Actual Budget v26.4.0, synced to a music track at ~139 BPM, using real Playwright screen recordings.

**Architecture:** The video is a single Remotion composition using `<TransitionSeries>` for beat-synced scene transitions. Each feature is a reusable `<FeatureScene>` component that wraps a screen recording in a browser frame with animated text overlays. Screen recordings are captured via Playwright against the running Actual Budget app with demo data.

**Tech Stack:** Remotion 4.0.443, React 19, Tailwind CSS 4, @remotion/transitions, @remotion/media, @remotion/google-fonts, Playwright (for screen capture)

- [ ] **Step 1: Create AnimatedText component**

This component animates text in with a spring pop effect, used by all scenes for titles and taglines.

```tsx
// src/components/AnimatedText.tsx
import { interpolate, spring, useCurrentFrame, useVideoConfig } from "remotion";

type AnimatedTextProps = {
  text: string;
  delay?: number;
  fontSize?: number;
  fontWeight?: string;
  color?: string;
  style?: React.CSSProperties;
};

export function AnimatedText({
  text,
  delay = 0,
  fontSize = 48,
  fontWeight = "bold",
  color = "#ffffff",
  style,
}: AnimatedTextProps) {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  const progress = spring({
    frame,
    fps,
    delay,
    config: { damping: 20, stiffness: 200 },
  });

  const translateY = interpolate(progress, [0, 1], [30, 0]);
  const opacity = interpolate(progress, [0, 1], [0, 1]);

  return (
    <div
      style={{
        fontSize,
        fontWeight,
        color,
        transform: `translateY(${translateY}px)`,
        opacity,
        textShadow: "0 2px 10px rgba(0,0,0,0.5)",
        ...style,
      }}
    >
      {text}
    </div>
  );
}
```

import {
  AbsoluteFill,
  Img,
  interpolate,
  spring,
  staticFile,
  useCurrentFrame,
  useVideoConfig,
} from "remotion";
import { loadFont } from "@remotion/google-fonts/Inter";
import { COLORS } from "../constants";

const { fontFamily } = loadFont("normal", {
  weights: ["400", "700", "900"],
  subsets: ["latin"],
});

export function TitleCard() {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  // Logo scales in with spring
  const logoScale = spring({
    frame,
    fps,
    config: { damping: 15, stiffness: 80, mass: 2 },
  });

  // Version text fades in after logo
  const versionOpacity = interpolate(frame, [1 * fps, 2 * fps], [0, 1], {
    extrapolateLeft: "clamp",
    extrapolateRight: "clamp",

  const whatsNewProgress = spring({
    frame,
    fps,
    delay: Math.round(2 * fps),

  const glowOpacity = interpolate(
    frame,

    [0, 0.3, 0.6],

    <AbsoluteFill
      style={{
        background: `linear-gradient(135deg, ${COLORS.bgDark}, ${COLORS.bgGradient})`,
        display: "flex",
        flexDirection: "column",
        alignItems: "center",

import {
  AbsoluteFill,
  interpolate,
  spring,
  staticFile,
  useCurrentFrame,
  useVideoConfig,
} from "remotion";
import { Video } from "@remotion/media";
import { loadFont } from "@remotion/google-fonts/Inter";
import { BrowserFrame } from "./BrowserFrame";
import { AnimatedText } from "./AnimatedText";
import { COLORS, FRAMES_PER_BEAT, type Feature } from "../constants";

const { fontFamily } = loadFont("normal", {
  weights: ["400", "700"],
  subsets: ["latin"],
});

type FeatureSceneProps = {
  feature: Feature;
};

export function FeatureScene({ feature }: FeatureSceneProps) {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  // Browser frame slides in from right on first beat
  const slideIn = spring({
    frame,
    fps,
    config: { damping: 20, stiffness: 200 },
  });
  const translateX = interpolate(slideIn, [0, 1], [200, 0]);

  // Subtle zoom on the recording for energy
  const zoom = interpolate(frame, [0, 5 * fps], [1, 1.05], {
    extrapolateRight: "clamp",
  });

  return (
    <AbsoluteFill
      style={{
        background: `linear-gradient(135deg, ${COLORS.bgDark}, ${COLORS.bgGradient})`,
        display: "flex",
        flexDirection: "column",
        alignItems: "center",
        justifyContent: "center",
        fontFamily,
      }}
    >
      {/* Accent glow behind browser frame */}
      <div
        style={{
          position: "absolute",
          width: 600,
          height: 400,
          background: `radial-gradient(ellipse, ${feature.accentColor}22, transparent)`,
          filter: "blur(60px)",
        }}
      />

      {/* Text overlay — top area */}
      <div
        style={{
          position: "absolute",
          top: 40,
          left: 0,
          right: 0,

                width: "100%",

            />
          </div>

}

import {
  AbsoluteFill,
  Img,
  interpolate,
  spring,
  staticFile,
  useCurrentFrame,
  useVideoConfig,
} from "remotion";
import { loadFont } from "@remotion/google-fonts/Inter";
import { COLORS, FRAMES_PER_BEAT } from "../constants";

const { fontFamily } = loadFont("normal", {
  weights: ["400", "700", "900"],
  subsets: ["latin"],
});

const STATS = [
  { number: "4", label: "Features" },
  { number: "45", label: "Enhancements" },
  { number: "32", label: "Bugfixes" },
];

export function OutroCard() {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  // Stats appear one by one
  const statElements = STATS.map((stat, i) => {
    const delay = i * FRAMES_PER_BEAT * 2;
    const progress = spring({
      frame,
      fps,
      delay,
      config: { damping: 20, stiffness: 200 },
    });
    const translateY = interpolate(progress, [0, 1], [40, 0]);

    return (
      <div
        key={stat.label}
        style={{
          display: "flex",
          flexDirection: "column",
          alignItems: "center",
          opacity: progress,
          transform: `translateY(${translateY}px)`,
        }}
      >
        <div
          style={{
            fontSize: 64,
            fontWeight: 900,
            color: COLORS.accentGold,
            lineHeight: 1,
          }}
        >
          {stat.number}
        </div>
        <div
          style={{
            fontSize: 20,
            color: COLORS.textSecondary,
            marginTop: 8,
            textTransform: "uppercase",
            letterSpacing: 2,
          }}
        >
          {stat.label}
        </div>
      </div>
    );
  });

  // CTA appears after stats
  const ctaDelay = STATS.length * FRAMES_PER_BEAT * 2 + FRAMES_PER_BEAT;
  const ctaProgress = spring({
    frame,
    fps,
    delay: ctaDelay,
    config: { damping: 200 },
  });

  // Fade out at the end
  const fadeOut = interpolate(
    frame,
    [durationInFrames - fps, durationInFrames],
    [1, 0],
    { extrapolateLeft: "clamp", extrapolateRight: "clamp" },
  );

  return (
    <AbsoluteFill
      style={{
        background: `linear-gradient(135deg, ${COLORS.bgDark}, ${COLORS.bgGradient})`,
        display: "flex",
        flexDirection: "column",
        alignItems: "center",
        justifyContent: "center",
        fontFamily,
        opacity: fadeOut,
      }}
    >
      {/* Background glow */}
      <div
        style={{
          position: "absolute",
          width: 500,
          height: 500,
          borderRadius: "50%",
          background: `radial-gradient(circle, ${COLORS.accentGold}22, transparent)`,
          filter: "blur(100px)",
        }}
      />

      {/* Stats row */}
      <div
        style={{
          display: "flex",
          gap: 80,
          marginBottom: 60,
        }}
      >
        {statElements}
      </div>

      {/* CTA */}
      <div
        style={{
          opacity: ctaProgress,
          transform: `translateY(${interpolate(ctaProgress, [0, 1], [20, 0])}px)`,
          display: "flex",
          flexDirection: "column",
          alignItems: "center",
          gap: 16,
        }}
      >
        <div
          style={{
            fontSize: 32,
            fontWeight: 700,
            color: COLORS.white,
          }}
        >
          Update now
        </div>
        <div
          style={{
            fontSize: 22,

import { AbsoluteFill, staticFile } from "remotion";
import { Audio } from "@remotion/media";

  TIER2_FEATURES,

export function MyComposition() {
  return (

      <TransitionSeries>
        {/* Title Card */}

            />

              key={`t1-scene-${i}`}
              durationInFrames={TIER1_SCENE_DURATION}
            >
              <FeatureScene feature={feature} />
            </TransitionSeries.Sequence>
          </>
        ))}

        {/* Tier 2: Quick highlights with slide transitions */}
        {TIER2_FEATURES.map((feature, i) => (
          <>
            <TransitionSeries.Transition
              key={`t2-transition-${i}`}
              presentation={slide({ direction: "from-right" })}
              timing={linearTiming({ durationInFrames: TRANSITION_DURATION })}
            />
            <TransitionSeries.Sequence
              key={`t2-scene-${i}`}
              durationInFrames={TIER2_SCENE_DURATION}
            >
              <FeatureScene feature={feature} />
            </TransitionSeries.Sequence>
          </>
        ))}

        {/* Outro with fade transition */}
        <TransitionSeries.Transition
          presentation={fade()}
          timing={linearTiming({ durationInFrames: TRANSITION_DURATION })}
        />
        <TransitionSeries.Sequence durationInFrames={OUTRO_DURATION}>
          <OutroCard />
        </TransitionSeries.Sequence>
      </TransitionSeries>
    </AbsoluteFill>
  );
}

// scripts/record-features.ts
import { chromium } from "playwright";
import path from "path";

const BASE_URL = process.env.BASE_URL || "http://localhost:3001";
const OUTPUT_DIR = path.resolve(__dirname, "../public/recordings");

async function delay(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

async function recordFeature(
  name: string,
  action: (page: import("playwright").Page) => Promise<void>,
) {
  const browser = await chromium.launch({ headless: false });
  const context = await browser.newContext({
    viewport: { width: 1024, height: 576 },
    recordVideo: {
      dir: OUTPUT_DIR,
      size: { width: 1024, height: 576 },
    },
  });

  const page = await context.newPage();
  await page.goto(BASE_URL);

  // Wait for app to load
  await page.waitForTimeout(3000);

  // Perform the feature action
  await action(page);

  // Let the recording capture the final state
  await delay(2000);

  await context.close();
  await browser.close();

  // Rename the recording (Playwright generates random names)
  const fs = await import("fs");
  const files = fs
    .readdirSync(OUTPUT_DIR)
    .filter((f: string) => f.endsWith(".webm"))
    .map((f: string) => ({
      name: f,
      time: fs.statSync(path.join(OUTPUT_DIR, f)).mtimeMs,
    }))
    .sort((a: { time: number }, b: { time: number }) => b.time - a.time);

  if (files.length > 0) {
    const latest = files[0].name;
    fs.renameSync(
      path.join(OUTPUT_DIR, latest),
      path.join(OUTPUT_DIR, `${name}.webm`),
    );
    console.log(`Saved: ${name}.webm`);
  }
}
